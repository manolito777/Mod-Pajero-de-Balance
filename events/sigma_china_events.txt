namespace = sigma_china_events

sigma_china_events.1 = { #Split china up into warlord states
	type = country_event
	placement = root
	duration = 3

	title = sigma_china_events.1.t
	desc = sigma_china_events.1.d
	flavor = sigma_china_events.1.f

	event_image = {
		video = "unspecific_politicians_arguing"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/five_races_flag.dds"

	trigger = {
		exists = c:CHI
		this = c:CHI

	}

	immediate = {

		if = {
			limit = { exists = s:STATE_FORMOSA.region_state:CHI }
			create_country = { 
				tag = FRS
				origin = this
				state = s:STATE_FORMOSA.region_state:CHI
				on_created = {
						1 = {
							set_strategy = ai_strategy_unify_china
						}
					}
				}
			}
		}
		option = {
		name = sigma_china_events.1.a
		add_modifier = {
			name = modifier_century_of_triumph_technology
			days = long_modifier_time
		}
		add_journal_entry = {
			type = taiwan_destruction_je
		}
	}
}
