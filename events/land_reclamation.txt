namespace = land_reclamation

land_reclamation.1 = {
	type = country_event
	placement = scope:land_reclamation_state

	title = land_reclamation.1.t
	desc = {
		first_valid = {
			triggered_desc = {
				desc = land_reclamation.1.d
				trigger = {
					OR = {
						AND = {
							NOT = { has_technology_researched = steel_frame_buildings }
							global_var:land_reclamation_global >= 9
						}
						AND = {
							has_technology_researched = steel_frame_buildings
							global_var:land_reclamation_global >= 8
						}
					}
				}
			}
			triggered_desc = {
				desc = land_reclamation.1.d2
				trigger = {
					always = yes
				}
			}
		}
	}
	flavor = land_reclamation.1.f

	event_image = {
		video = "europenorthamerica_russian_serfs"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_map.dds"

	duration = 3

	trigger = {
		# triggered by JE
	}

	immediate = {
		}

	option = {
		name = land_reclamation.1.a
		default_option = yes
		trigger = {
			NOT = { has_technology_researched = steel_frame_buildings }
		}
		if = {
			limit = {
				global_var:land_reclamation_global >= 9
			}
			every_scope_state = {
			limit = {
				OR = {
					state_region = s:STATE_HOLLAND
					state_region = s:STATE_FRIESLAND
					state_region = s:STATE_GELRE
				}
			}
			add_modifier = {
				name = modifier_reclamation_polder
				days = -1      # permanente
			}
		}
		}
		change_global_variable = {
			name = land_reclamation_global
			add = 1
		}
		every_scope_state = {
			limit = {
				OR = {
					state_region = s:STATE_HOLLAND
					state_region = s:STATE_FRIESLAND
					state_region = s:STATE_GELRE
				}
			}
			state_region = {
				add_arable_land = 1
			}
		}
	}

	option = {
		name = land_reclamation.1.b
		default_option = yes
		highlighted_option = yes
		trigger = {
			has_technology_researched = steel_frame_buildings
		}
		if = {
			limit = {
				global_var:land_reclamation_global >= 8
			}
			every_scope_state = {
			limit = {
				OR = {
					state_region = s:STATE_HOLLAND
					state_region = s:STATE_FRIESLAND
					state_region = s:STATE_GELRE
				}
			}
			add_modifier = {
				name = modifier_reclamation_polder
				days = -1      # permanente
			}
		}
		}
		if = {
			limit = {
				global_var:land_reclamation_global <= 8
			}
			change_global_variable = {
				name = land_reclamation_global
				add = 2
			}
			every_scope_state = {
			limit = {
				OR = {
					state_region = s:STATE_HOLLAND
					state_region = s:STATE_FRIESLAND
					state_region = s:STATE_GELRE
				}
			}
				state_region = {
					add_arable_land = 3
				}
			}
		}
		else = {
			change_global_variable = {
				name = river_tigris_reclamation_global
				add = 2
			}
			every_scope_state = {
			limit = {
				OR = {
					state_region = s:STATE_HOLLAND
					state_region = s:STATE_FRIESLAND
					state_region = s:STATE_GELRE
				}
			}
				state_region = {
					add_arable_land = 1
				}
			}
		}
		
	}
}
