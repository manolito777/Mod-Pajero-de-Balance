namespace = revrusa

revrusa.1 = {
	type = country_event
	placement = ROOT

	title = revrusa.1.t
	desc = revrusa.1.d
	flavor = revrusa.1.f

	event_image = {
		video = "unspecific_temperance_movement"
	}

	icon = "gfx/interface/icons/event_icons/event_hellenic.dds"

	duration = 3

	trigger = {
	}

	immediate = {
		random_scope_state = {
			limit = {
				owner = ROOT
				is_incorporated = yes
				NOT = { has_modifier = estado_asegurado }
				NOT = { is_capital = yes }
			}
			save_scope_as = revrusa_spawn_state
		}

		create_country = {
			tag = WTH
			origin = ROOT
			state = scope:revrusa_spawn_state
		}
		
		c:WTH = { 
			save_scope_as = wth_scope 
			# Limpiamos diplomacia para asegurar que pueda iniciar la guerra
			diplomacy_cleanup = yes
		}

		every_scope_state = {
			limit = {
				owner = ROOT
				NOT = { has_modifier = estado_asegurado }
				NOT = { is_capital = yes }
				NOT = { state_region = scope:revrusa_spawn_state.state_region }
			}
			set_state_owner = c:WTH
			set_state_type = incorporated
		}
	}

option = {
		name = revrusa.1.a
		default_option = yes
		
		c:RUS ?= {
				create_character = {
				last_name = Lenin
				historical = yes
				culture = cu:russian
				birth_date = 1815.4.29
				interest_group = ig_trade_unions
				ideology = moderate
				traits = {
					meticulous
					romantic
					imperious
				}
				on_created = {
								set_character_as_ruler = yes
							}
			}
}
		
		activate_law = law_type:law_soviet_economy
		activate_law = law_type:law_supreme_soviet
		activate_law = law_type:law_council_of_ministers
		
		ig:ig_devout ?= {
			remove_ideology = ideology_moralist 
			add_ideology = ideology_moralist_socialism
		}
		ig:ig_trade_unions ?= {
			remove_ideology = ideology_populist
			add_ideology = ideology_stalinismo
			remove_ideology = ideology_egalitarian
			add_ideology = ideology_KGB
		}
		ig:ig_armed_forces ?= {
			add_ideology = ideology_socialist
		}
		scope:wth_scope = {
			change_infamy = -500 
			
			create_diplomatic_play = {
				name = russia_rebellion
				target_country = ROOT
				type = dp_annex_war
				add_war_goal = {
					holder = c:RUS
					target_country = c:WTH
					type = annex_country
				}
			}
		}
		
	}
}
