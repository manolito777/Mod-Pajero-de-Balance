    icon = "gfx/interface/icons/invention_icons/urbanization.dds"
    group = je_group_technology

    # opcional: solo para un país con cultura primaria rusa
    possible = {
        country_has_primary_culture = cu:russian
    }

    immediate = {
        random_scope_state = {
            limit = { state_region = s:STATE_OB }
            save_scope_as = je_ob_state
        }
        random_scope_state = {
            limit = { state_region = s:STATE_SURGUT }
            save_scope_as = je_surgut_state
        }
        cu:russian = { save_scope_as = je_russian_culture }
    }

    complete = {
            any_scope_state = {
                state_region = s:STATE_OB
                culture_percent_state = {
                    target = cu:russian
                    value >= 0.5
                }
            }
            any_scope_state = {
                state_region = s:STATE_SURGUT
                culture_percent_state = {
                    target = cu:russian
                    value >= 0.5
                }
            }
        }

    on_complete = {
        trigger_event = { id = revrusa.9 }
    }

    weight = 999
    should_be_pinned_by_default = yes
}
