je_valfardsstaten = {
	icon = "gfx/interface/icons/invention_icons/urbanization.dds"
	group = je_group_technology

	is_shown_when_inactive = {
	    owns_entire_state_region = STATE_SVEALAND
		country_has_primary_culture = cu:swedish
	}

	possible = {
	}

	complete = {
		s:STATE_SVEALAND = {
		    any_scope_state = {
				any_scope_building = {
					is_building_type = building_urban_center
					level >= 100
				}
			}
		}
		s:STATE_NORRLAND = {
		    any_scope_state = {
				any_scope_building = {
					is_building_type = building_urban_center
					level >= 20
				}
			}
		}
		s:STATE_SCANIA = {
		    any_scope_state = {
				any_scope_building = {
					is_building_type = building_urban_center
					level >= 20
				}
			}
		}
		s:STATE_GOTALAND = {
		    any_scope_state = {
				any_scope_building = {
					is_building_type = building_urban_center
					level >= 20
				}
			}
		}
	}
	
	on_complete = {
		trigger_event = { id = valfardsstaten_event.1 }
	}
	weight = 50
	should_be_pinned_by_default = yes
}
