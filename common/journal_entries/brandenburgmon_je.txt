brandenburgmon_je = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	group = je_group_technology

	on_monthly_pulse = {
		effect = {
			change_variable = {
				name = bran_var
				add = 1
			}
		}
	}

	complete = {
		scope:journal_entry = { is_goal_complete = yes }
		any_scope_state = {
		    state_region = s:STATE_BRANDENBURG
			is_incorporated = yes
			has_modifier = BerlinReformed
			}
		}
	}

	on_complete = {
		remove_modifier = branmon_survey
		trigger_event = { id = branmon.1 } # site found
	}

	invalid = {
		NOT = {
			any_scope_state = {
			state_region = s:STATE_BRANDENBURG
			is_incorporated = yes
			has_modifier = BerlinReformed
				}
			}
		}
	}

	on_invalid = {
		remove_modifier = branmon_survey
		remove_variable = bran_var
	}

	current_value = {
		value = root.var:bran_var
	}

	goal_add_value = {
		value = 24
	}

	progressbar = yes

	weight = 10

	should_be_pinned_by_default = yes
}

brandenburgmon_je2 = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	group = je_group_technology

	complete = {
		scope:target = {
			any_scope_building = {
				is_building_type = building_branmon
				is_under_construction = no
			}
		}
	}

	fail = {
		NOT = {
			any_scope_state = {
				has_modifier = branmon_site
			}
		}
	}

	on_fail = {
		remove_variable = bran_var
	}

	on_complete = {
		trigger_event = { id = branmon.2 }
		remove_variable = bran_var
		scope:target = {
			remove_modifier = branmon_site
		}
	}

	should_be_pinned_by_default = yes
}
