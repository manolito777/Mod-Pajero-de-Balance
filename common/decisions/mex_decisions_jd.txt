	}

	possible = {
		this = c:MEX

		c:GBR = {
			is_at_war = no
			is_subject = no
			NOT = { owes_obligation_to = ROOT }
			relations:root >= relations_threshold:cordial
		}
		hidden_trigger = {
			NOT = { has_variable = buy_gbr_colonies }
		}
	}

	when_taken = {
		set_variable = {
			name = buy_gbr_colonies
			value = yes
		}
		custom_tooltip = {
			text = buy_gbr_colonies_decision_tooltip
		}
		annex = c:MKT
		c:MEX = {
			change_relations = {
				country = c:GBR
				value = 20
			}
		}
		s:STATE_GUATEMALA  = {
			random_scope_state = {
				limit = {
					exists = c:GBR
					owner = {
						OR = {
							c:GBR = THIS
							is_subject_of = c:GBR
						}
					}
				}
				set_state_owner = c:MEX
			}
		}
		add_treasury = -500000
	}

	ai_chance = {
		value = 3
	}
}
