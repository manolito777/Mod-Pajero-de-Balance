brandenburgmon_decision = {
	is_shown = {
		has_technology_researched = steel_frame_buildings
		any_scope_state = {
			state_region = s:STATE_BRANDENBURG
			is_incorporated = yes
			
		}

	}

	possible = {
		produced_bureaucracy > 500
		any_scope_state = {

			state_region = s:STATE_BRANDENBURG
			is_incorporated = yes
			has_modifier = BerlinReformed
			}
		}
	

	when_taken = {
		set_variable = {
			name = bran_var
			value = 0
		}
		add_modifier = {
			name = skyscraper_survey
			months = 24
		}
		add_journal_entry = {
			type = brandenburgmon_je
		}
	}

	ai_chance = {
		value = 0
		if = {
			limit = { produced_bureaucracy > 2500 }
			add = 5
		}
		if = {
			limit = {
				produced_bureaucracy > 5000
				any_country = {
					is_ai = no
					has_variable = bran_var
				}
			}
			add = 50
		}
		if = {
			limit = {
				OR = {
					is_diplomatic_play_committed_participant = yes
					is_at_war = yes
				}
			}
			multiply = 0
		}
	}
}
