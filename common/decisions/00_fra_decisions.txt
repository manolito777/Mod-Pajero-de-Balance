#Plan Haussmann.
institute_plan_haussmann = {
	is_shown = {
		any_scope_state = {
			state_region = s:STATE_ILE_DE_FRANCE
			is_incorporated = yes
			has_modifier = old_paris
		}
	}

	possible = {
		bureaucracy >= 1500

		has_technology_researched = steel_frame_buildings
	}

	when_taken = {

		random_scope_state = {
			limit = {
				state_region = s:STATE_ILE_DE_FRANCE
			}
			remove_modifier = old_paris
			add_modifier = {
				name = new_paris
			}	
		}

		add_modifier = {
			name = rebuilding_paris
			years = 5
		}
	}

	ai_chance = {
		base = 4
	}
}
